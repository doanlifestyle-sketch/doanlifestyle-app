const VOICES = [...BASE_VOICES, ...Array.from({ length: 40 }, (_, i) => ({name: `Voice ${i + 11}`,voice: CORE[i % CORE.length],style: EXTRA_STYLES[i % EXTRA_STYLES.length]}))];
let selectedVoice = "alloy";
let selectedStyle = PRESETS.story.style;
let currentAudioBlob = null;

function setStatus(msg){ $("status").textContent = msg; }
function updateCounter(){ $("counter").textContent = $("text").value.length.toLocaleString('vi-VN') + " ký tự"; }
function updateSpeed(){ $("speedValue").textContent = Number($("speed").value).toFixed(2); }
function safeName(v){ return (v || "doanlifestyle_tts_studio").trim().replace(/[^\w\-]+/g,"_"); }

function renderPresets(){
  const wrap = $("presetPills"); wrap.innerHTML = "";
  Object.keys(PRESETS).forEach(key => {
    const p = PRESETS[key];
    const btn = document.createElement("button");
    btn.className = "pill" + (p.voice === selectedVoice && p.style === selectedStyle ? " active" : "");
    btn.textContent = p.label;
    btn.onclick = () => { selectedVoice = p.voice; selectedStyle = p.style; $("instructions").value = p.style; renderPresets(); renderVoices(); setStatus(`Đã chọn preset: ${p.label}`); };
    wrap.appendChild(btn);
  });
}

function renderVoices(){
  const container = $("voiceGrid"); container.innerHTML = "";
  VOICES.forEach((v) => {
    const card = document.createElement("div");
    card.className = "voiceCard" + (v.voice === selectedVoice && v.style === selectedStyle ? " active" : "");
    card.innerHTML = `
      <div class="voiceName">${v.name}</div>
      <div class="voiceMeta">${v.voice} • ${v.style}</div>
      <div class="voiceActions">
        <button class="smallBtn blue" type="button">▶️ Nghe thử</button>
        <button class="smallBtn green" type="button">✅ Chọn</button>
      </div>
    `;
    const [previewBtn, selectBtn] = card.querySelectorAll("button");
    previewBtn.onclick = () => previewVoice(v.voice, v.style);
    selectBtn.onclick = () => { selectedVoice = v.voice; selectedStyle = v.style; $("instructions").value = v.style; renderVoices(); renderPresets(); setStatus(`Đã chọn: ${v.name}`); };
    container.appendChild(card);
  });
}

async function createSpeechBlob(inputText, voice, style){
  const apiKey = $("apiKey").value.trim();
  if(!apiKey) throw new Error("Bạn chưa nhập OpenAI API key.");
  const payload = {model: $("model").value,voice: voice,input: inputText,format: $("format").value,response_format: $("format").value,speed: Number($("speed").value),instructions: style};
  const res = await fetch("https://api.openai.com/v1/audio/speech", {method: "POST",headers: {"Authorization": "Bearer " + apiKey,"Content-Type": "application/json"},body: JSON.stringify(payload)});
  if(!res.ok){ const detail = await res.text().catch(() => ""); throw new Error(res.status + "\n" + detail); }
  return await res.blob();
}

async function previewVoice(voice, style){
  try{
    setStatus("Đang tạo đoạn nghe thử...");
    const blob = await createSpeechBlob("Xin chào, đây là giọng đọc thử để bạn cảm nhận.", voice, style);
    const url = URL.createObjectURL(blob);
    const audio = new Audio(url);
    audio.play();
    setStatus(`Đang nghe thử: ${voice}`);
  }catch(err){ setStatus("Lỗi nghe thử:\n" + err.message); }
}

async function generateAudio(){
  const text = $("text").value.trim();
  if(!text){ setStatus("Bạn chưa nhập nội dung."); return; }
  try{
    setStatus("Đang tạo audio...");
    const style = $("instructions").value.trim() || selectedStyle;
    const blob = await createSpeechBlob(text, selectedVoice, style);
    currentAudioBlob = blob;
    if(currentAudioUrl) URL.revokeObjectURL(currentAudioUrl);
    currentAudioUrl = URL.createObjectURL(blob);
    $("player").src = currentAudioUrl;
    setStatus(`Đã tạo audio thành công với ${selectedVoice}.`);
  }catch(err){ setStatus("Không tạo được audio.\n" + err.message); }
}

function downloadAudio(){
  if(!currentAudioBlob || !currentAudioUrl){ setStatus("Chưa có audio để tải."); return; }
  const a = document.createElement("a");
  a.href = currentAudioUrl;
  a.download = `${safeName($("filename").value)}.${$("format").value}`;
  document.body.appendChild(a);
  a.click();
  a.remove();
}

  currentAudioUrl = null; currentAudioBlob = null;
  $("player").removeAttribute("src"); $("player").load();
  updateCounter(); updateSpeed(); renderVoices(); renderPresets(); setStatus("Đã đặt lại.");
}
